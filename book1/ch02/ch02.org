* Processing a Guess

#+begin_src

use std::io; // import

fn main() {
    println!("Guess the number!");

    println!("Please input your guess.");

    let mut guess = String::new();

    io::stdin()
        .read_line(&mut guess) // returns Result type (Ok | Err)
        .expect("Failed to read line"); // if Result is Err display msg and crash program
                                        // if Result is Ok return value in Ok (the number of bytes typed by user)

    println!("You guessed: {guess}");
}

#+end_src
